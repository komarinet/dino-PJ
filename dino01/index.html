const ctx = canvas.getContext('2d'); // 恐竜の最初のステータス（位置と大きさ） let dino = { x: 130, y: 130, size: 40, emoji: '🦖' }; // 画面に恐竜を描くお仕事（関数） function draw() { // 一度画面をまっさらに消すの（残像が残らないようにね） ctx.clearRect(0, 0, canvas.width, canvas.height); // 恐竜の絵文字を描くわよ！ ctx.font = dino.size + "px sans-serif"; ctx.fillText(dino.emoji, dino.x, dino.y + dino.size); } // スマホでタップした時の動きよ canvas.addEventListener('touchstart', function(e) { e.preventDefault(); // 画面がスクロールしちゃうのを防ぐおまじない const rect = canvas.getBoundingClientRect(); const touch = e.touches[0]; // タップした場所に恐竜を移動させる計算よ dino.x = touch.clientX - rect.left - dino.size / 2; dino.y = touch.clientY - rect.top - dino.size / 2; draw(); }, { passive: false }); // パソコンの矢印キーを押した時の動きよ window.addEventListener('keydown', function(e) { if(e.key === 'ArrowUp') dino.y -= 15; if(e.key === 'ArrowDown') dino.y += 15; if(e.key === 'ArrowLeft') dino.x -= 15; if(e.key === 'ArrowRight') dino.x += 15; draw(); }); // 最初に1回だけ恐竜を描画してあげるの draw();
